-- 36
SELECT deptno, COUNT(*) AS employees_with_manager
FROM emp
WHERE mgr IS NOT NULL
GROUP BY deptno
HAVING COUNT(*) > (
  SELECT AVG(cnt)
  FROM (
    SELECT COUNT(*) AS cnt
    FROM emp
    WHERE mgr IS NOT NULL
    GROUP BY deptno
  ) t
);

-- 73 (FULL OUTER JOIN simulation)
SELECT e.empno, e.ename, d.deptno, d.dname
FROM emp e
LEFT JOIN dept d ON e.deptno = d.deptno

UNION

SELECT e.empno, e.ename, d.deptno, d.dname

-- 74 (Managers using self join)
SELECT DISTINCT m.ename AS manager
FROM emp e
JOIN emp m ON e.mgr = m.empno;

-- 75 (CROSS JOIN)
SELECT e.ename, d.dname
FROM emp e
CROSS JOIN dept d;

-- 76 (EXISTS)
SELECT d.deptno, d.dname
FROM dept d
WHERE EXISTS (
    SELECT 1 
    FROM emp e 
    WHERE e.deptno = d.deptno AND e.sal > 2500
);

-- 77 (NOT EXISTS)
SELECT d.dname, d.deptno
FROM dept d
WHERE NOT EXISTS (
    SELECT 1 
    FROM emp e 
    WHERE e.deptno = d.deptno AND e.sal < 1000
);

-- 78 (Managers in New York)
SELECT m.ename AS manager_name, COUNT(e.empno) AS emp_count
FROM emp e
JOIN emp m ON e.mgr = m.empno
JOIN dept d ON e.deptno = d.deptno
WHERE d.location = 'New York'
GROUP BY m.ename;

-- 79 (FULL JOIN with location)

RIGHT JOIN dept d ON e.deptno = d.deptno;
